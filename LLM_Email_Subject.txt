# Step 1: Create your prompts

system_prompt = "You are an assistant that generates an appropriate short subject line for an email based on its content."

user_prompt = """
    Dear team,

    I wanted to update you on the status of the project. We have successfully completed the first phase and are now ready to move forward with the second phase. The deadline for phase two is next Friday, so please ensure all tasks are completed by then.

    Please let me know if you have any questions or concerns.
"""

# Step 2: Make the messages list

messages = [
    {"role": "system", "content": system_prompt},
    {"role": "user", "content": user_prompt}
]

# Step 3: Call OpenAI

response = openai.chat.completions.create(model="gpt-4o-mini", messages=messages)


# Step 4: print the result

print(response.choices[0].message.content)
